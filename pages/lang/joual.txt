  function parseShareParams() {
    var params = new URLSearchParams(window.location.search);
    return {
      fr: params.get('fr') || null,
      en: params.get('en') || null,
      tr: params.get('tr') || null,
      jo: params.get('jo') || null,
      cu: params.has('cu'),
      cr: params.has('cr'),
      ex: params.has('ex')
    };
  }
  var shareParams = parseShareParams();

  function applyShareParams(enInputEl) {
    if (!shareParams) return false;
    if (chkCute) chkCute.checked = !!shareParams.cu;
    if (chkGrackle) chkGrackle.checked = !!shareParams.cr;
    if (chkExpr) chkExpr.checked = !!shareParams.ex;
    // New format: fr= has formal French, en= has original English (display only)
    var frenchText = shareParams.fr || shareParams.jo || shareParams.tr;
    if (!frenchText) return false;
    // Populate English box if present (just for context, no translation triggered)
    var engText = shareParams.en || (shareParams.tr ? shareParams.tr : null);
    if (enInputEl && engText) enInputEl.value = engText;
    // Put formal French into the input and trigger joual conversion
    E.resetOverrides();
    inputEl.value = frenchText;
    doTranslate();
    return true;
  }

    if (btnExample) {
      btnExample.addEventListener('click', function() {
        var pick = EXAMPLES[Math.floor(Math.random() * EXAMPLES.length)];
        inputEl.value = pick;
        if (enInputEl) enInputEl.value = '';
        E.resetOverrides();
        doTranslate();
        btnExample.textContent = '🎯 ✓';
        setTimeout(function(){ btnExample.textContent = '🎯 Example'; }, 1200);
      });
    }

    if (btnClearInput) {
      btnClearInput.addEventListener('click', function() {
        inputEl.value = '';
        if (enInputEl) enInputEl.value = '';
        E.resetOverrides();
        doTranslate();
      });
    }

    // Copy button handler
    var copyBtn = document.getElementById('copy-output-btn');

    // Shuffle button — randomize alternatives for all substitution spans
    var btnRandomize = document.getElementById('btn-randomize');
    if (btnRandomize) {
      btnRandomize.addEventListener('click', function() {
        var spans = outputEl.querySelectorAll('.joual-sub');
        if (!spans.length) return;
        var changed = 0;
        for (var i = 0; i < spans.length; i++) {
          var sp = spans[i];
          var altsJson = sp.getAttribute('data-alts');
          if (!altsJson) continue;
          try { var alts = JSON.parse(altsJson); } catch(e) { continue; }
          if (!alts || alts.length < 2) continue;
          // Pick a random alt (up to 5 choices considered)
          var pool = alts.slice(0, Math.min(alts.length, 5));
          var pick = pool[Math.floor(Math.random() * pool.length)];
          var newText = E.matchCase(sp.getAttribute('data-original') || '', pick.text);
          if (newText !== sp.textContent) {
            sp.textContent = newText;
            sp.setAttribute('data-current', newText);
            // Update override so subsequent translates remember this choice
            var entryIdx = sp.getAttribute('data-entry-idx');
            var matchStart = sp.getAttribute('data-match-start');
            if (entryIdx && matchStart) E.setOverride(entryIdx + '-' + matchStart, newText);
            changed++;
          }
        }
        if (changed > 0) {
          btnRandomize.textContent = '🎲 ' + changed + '!';
          setTimeout(function(){ btnRandomize.textContent = '🎲 Shuffle'; }, 1200);
        }
      });
    }

    if (copyBtn) {
      copyBtn.addEventListener('click', function() {
        var text = outputEl.innerText || outputEl.textContent || '';
        if (!text.trim() || outputEl.classList.contains('placeholder')) return;
        if (navigator.clipboard && navigator.clipboard.writeText) {
          navigator.clipboard.writeText(text).then(function() {
            copyBtn.textContent = '✓ Copied!';
            setTimeout(function() { copyBtn.textContent = '📋 Copy'; }, 1500);
          });
        } else {
          // Fallback for older browsers
          var ta = document.createElement('textarea');
          ta.value = text;
          ta.style.position = 'fixed';
          ta.style.opacity = '0';
          document.body.appendChild(ta);
          ta.select();
          document.execCommand('copy');
          document.body.removeChild(ta);
          copyBtn.textContent = '✓ Copied!';
          setTimeout(function() { copyBtn.textContent = '📋 Copy'; }, 1500);
        }
      });
    }

    // Apply default language (English-first)
    try { applyLanguage('en'); } catch (e) { /* fail silently if DOM different */ }
  }

  function updateOutputLabel() {
    var base = document.getElementById('joual-output-label');
    if (!base) return;
    var parts = ['Joual'];
    if (chkCute && chkCute.checked) parts.push('Cute');

    base.textContent = (currentLang === 'fr') ? frLabel : label;

  function doTranslate() {
    var text = inputEl.value;
    evaluateModeAvailability(text);
    var html = E.translate(text, {
      simplify: chkSimplify.checked,
      cute: chkCute.checked,
      grackle: chkGrackle.checked,
      expr: chkExpr.checked
    });
    if (html && html.trim()) {
      outputEl.classList.remove('placeholder');
      outputEl.innerHTML = html;
    } else {
      outputEl.classList.add('placeholder');
      outputEl.innerHTML = 'See Joual output appear here automatically\n\nExample:\nY\'a pas un chat dans la cabane.';
    }
    updateStats(text);
  }

  /**
   * Update the stats bar.
   */
  function updateStats(text) {
    if (!statsEl) return;
    if (!text || !text.trim()) {
      statsEl.textContent = "";
      return;
    }
    var words = text.trim().split(/\s+/).length;
    var subs = outputEl.querySelectorAll(".joual-sub").length;
    var mode = [];
    var names = (currentLang === 'fr') ? { simplify: 'simplifié', cute: 'extra-cute', grackle: 'critter-ize', expr: 'expressions' }
                                         : { simplify: 'simplified', cute: 'extra-cute', grackle: 'critter-ize', expr: 'expressions' };
    if (chkSimplify && chkSimplify.checked) mode.push(names.simplify);
    if (chkCute && chkCute.checked) mode.push(names.cute);
    if (chkGrackle && chkGrackle.checked) mode.push(names.grackle);
    if (chkExpr && chkExpr.checked) mode.push(names.expr);
    var wordsLabel = (currentLang === 'fr') ? 'mots' : 'words';
    var subsLabel = (currentLang === 'fr') ? 'substitution' : 'substitution';
    if (currentLang !== 'fr') {
      // English plural handling
      subsLabel = subs === 1 ? 'substitution' : 'substitutions';
    } else {
      subsLabel = subs === 1 ? 'substitution' : 'substitutions';
    }
    statsEl.textContent = words + ' ' + wordsLabel + ' · ' + subs + ' ' + subsLabel
      + (mode.length ? ' · mode: ' + mode.join(', ') : '');
  }

  function getProcessedInputForModes(text) {
    var value = text || "";
    if (chkSimplify && chkSimplify.checked) {
      value = E.stripAccents(value);
      value = E.normalizePunctuation(value);
    }
    return value;
  }

  function evaluateModeAvailability(text) {
    if (!chkCute || !chkGrackle) return;
    var processed = getProcessedInputForModes(text);
    var cuteAvailable = hasCuteModeImpact(processed);
    var grackleAvailable = hasGrackleModeImpact(processed);
    if (!cuteAvailable && chkCute.checked) chkCute.checked = false;
    if (!grackleAvailable && chkGrackle.checked) chkGrackle.checked = false;
    chkCute.disabled = !cuteAvailable;
    chkGrackle.disabled = !grackleAvailable;
    updateModeTooltips(cuteAvailable, grackleAvailable);
  }

  function updateModeTooltips(cuteAvailable, grackleAvailable) {
    var hints = (modeHints[currentLang] || modeHints.en) || {};
    if (labelCute) {
      var hint = cuteAvailable ? hints.cuteEnabled : hints.cuteDisabled;
      var parts = [];
      if (defaultCuteTitle) parts.push(defaultCuteTitle);
      if (hint) parts.push(hint);
      if (parts.length) labelCute.setAttribute('title', parts.join(' · '));
      else labelCute.removeAttribute('title');
    }
    if (labelGrackle) {
      var hint = grackleAvailable ? hints.grackleEnabled : hints.grackleDisabled;
      var parts = [];
      if (defaultGrackleTitle) parts.push(defaultGrackleTitle);
      if (hint) parts.push(hint);
      if (parts.length) labelGrackle.setAttribute('title', parts.join(' · '));
      else labelGrackle.removeAttribute('title');
    }
  }

  function hasCuteModeImpact(text) {
    if (!text || !text.trim()) return false;
    for (var i = 0; i < entries.length; i++) {
      var entry = entries[i];
      if (!entry.cute || !entry.cute.length) continue;
      var base = (entry.joual && entry.joual.length) ? entry.joual[0] : (entry.standard || "");
      var cuteFirst = entry.cute[0] || "";
      if (!base || !cuteFirst) continue;
      if (cuteFirst.toLowerCase() === base.toLowerCase()) continue;
      entry.regex.lastIndex = 0;
      if (entry.regex.test(text)) return true;
    }
    for (var j = 0; j < CUTE_CONTRACTIONS.length; j++) {
      var pattern = CUTE_CONTRACTIONS[j].pattern;
      if (!pattern) continue;
      pattern.lastIndex = 0;
      if (pattern.test(text)) return true;
    }
    return false;
  }

  function hasGrackleModeImpact(text) {
    if (!text || !text.trim()) return false;
    for (var i = 0; i < entries.length; i++) {
      var entry = entries[i];
      if (!entry.grackle || !entry.grackle.length) continue;
      entry.regex.lastIndex = 0;
      if (entry.regex.test(text)) return true;
    }
    return false;
  }

  var tooltipHideTimer = null;

  function showTooltip(span, evt) {
    clearTimeout(tooltipHideTimer);

    var original  = span.getAttribute("data-original") || "";
    var english   = span.getAttribute("data-english") || "";
    var standard  = span.getAttribute("data-standard") || "";
    var current   = span.getAttribute("data-current") || "";
    var entryIdx  = span.getAttribute("data-entry-idx") || "0";
    var matchStart= span.getAttribute("data-match-start") || "0";
    var altsJson  = span.getAttribute("data-alts") || "[]";

    var alts = [];
    try { alts = JSON.parse(altsJson); } catch(e) {}

    var html = '<div class="tt-header">' + E.escapeHtml(standard) + ' → ' + E.escapeHtml(current) + '</div>';
    if (english) {
      var contracted = E.englishContract(english || '');
      var engHtml = E.escapeHtml(english);
      if (contracted && contracted.toLowerCase() !== (english||'').toLowerCase()) {
        engHtml += ' → ' + E.escapeHtml(contracted);
      }
      html += '<div class="tt-english">' + engHtml + '</div>';
    }

    if (alts.length > 1) {
      html += '<div class="tt-alts-label">Alternatives :</div><div>';
      for (var i = 0; i < alts.length; i++) {
        var isActive = (alts[i].text.toLowerCase() === current.toLowerCase()) ? ' tt-alt-active' : '';
        var modeTag = '';
        if (alts[i].mode === 'cute') modeTag = ' (cute)';
        else if (alts[i].mode === 'grackle') modeTag = ' (critter)';
        else if (alts[i].mode === 'standard') modeTag = ' (std)';

        html += '<span class="tt-alt' + isActive + '"'
          + ' data-entry-idx="' + E.escapeAttr(entryIdx) + '"'
          + ' data-match-start="' + E.escapeAttr(matchStart) + '"'
          + ' data-alt-text="' + E.escapeAttr(alts[i].text) + '"'
          + '>' + E.escapeHtml(alts[i].text) + modeTag + '</span>';
      }
      html += '</div>';
    }

    tooltipEl.innerHTML = html;
    tooltipEl.style.display = "block";

    // Position near the span
    var rect = span.getBoundingClientRect();
    var appRect = document.getElementById("joual-app").getBoundingClientRect();
    var ttLeft = rect.left - appRect.left;
    var ttTop = rect.bottom - appRect.top + 6;

    // Keep within viewport
    tooltipEl.style.left = ttLeft + "px";
    tooltipEl.style.top = ttTop + "px";

    // After rendering, adjust if off-screen
    requestAnimationFrame(function() {
      var ttRect = tooltipEl.getBoundingClientRect();
      if (ttRect.right > window.innerWidth - 10) {
        tooltipEl.style.left = Math.max(0, ttLeft - (ttRect.right - window.innerWidth + 20)) + "px";
      }
      if (ttRect.bottom > window.innerHeight - 10) {
        tooltipEl.style.top = (rect.top - appRect.top - ttRect.height - 6) + "px";
      }
    });
  }

  function hideTooltip() {
    clearTimeout(tooltipHideTimer);
    tooltipEl.style.display = "none";
  }

  function hideTooltipDelayed() {
    clearTimeout(tooltipHideTimer);
    tooltipHideTimer = setTimeout(hideTooltip, 250);

===== About this Translator =====

Joual is a Québec French sociolect noted for contractions, anglicisms, softened sacres, and a distinctive phonetic flavour. This translator covers:

  * Common contractions (je → j', tu → t', il → y, etc.) and Québec idioms
  * 🐾 Critter-ize: critter-themed vocabulary for roleplay (grackles, rats, ducks, cheese, worms, etc.) with animal-themed alternatives to common words
  * Engine uses a longest-match-first algorithm with tokenization to avoid cascading replacements
  * Mild sacres included are softened variants (tabarnouche, câline, simonac)
  * Dictionary based on common Québec French, Montréal-era joual, and critter/RP expressions
  * English-to-French translation provided by [[https://www.deepl.com/en/docs-api|DeepL API Free]]
