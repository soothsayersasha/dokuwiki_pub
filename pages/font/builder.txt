<script>
let state = { main: null, top: [], side: [] };
let activeTarget = null; 

function toggleYivBuilderFont() {
    let wrapper = document.getElementById("yiv-builder-wrapper");
    let modal = document.getElementById("glyph-picker-modal");
    wrapper.classList.toggle("italic-mode");
    modal.classList.toggle("italic-mode");
}

function triggerMainHint() {
    let mainBtn = document.getElementById('main-drop-zone');
    if (mainBtn) {
        mainBtn.classList.remove('throb-hint');
        void mainBtn.offsetWidth; // Trigger reflow
        mainBtn.classList.add('throb-hint');
    }
}

renderUI();

function renderUI() {
    renderMain();
    renderTop();
    renderSide();
    updateOutput();
}

function createBox(glyphObj, area, index, sizeClass, isMain = false) {
    let charDisplay = glyphObj.char + (glyphObj.reversed ? '!' : '');
    let box = document.createElement('div');
    box.className = `glyph-box yiv-font ${sizeClass}`;
    if (isMain) box.id = 'main-drop-zone';
    box.innerText = charDisplay;
    if (isMain) box.onclick = (e) => { if(e.target === box) openPicker('main', 'set'); };

    let btnX = document.createElement('div');
    btnX.className = 'ctrl-btn ctrl-x';
    btnX.innerHTML = '✕';
    btnX.onclick = (e) => { e.stopPropagation(); removeGlyph(area, index); };
    box.appendChild(btnX);

    let btnFlip = document.createElement('div');
    btnFlip.className = `ctrl-btn ${area === 'top' ? 'ctrl-flip-top' : 'ctrl-flip-bot'}`;
    btnFlip.innerHTML = '⟳';
    btnFlip.onclick = (e) => { e.stopPropagation(); flipGlyph(area, index); };
    box.appendChild(btnFlip);

    return box;
}

function renderMain() {
    const container = document.getElementById('area-main');
    container.innerHTML = '';
    if (!state.main) {
        let btn = document.createElement('div');
        btn.id = 'main-drop-zone';
        btn.className = 'glyph-box main-box';
        btn.style.borderStyle = 'dashed';
        btn.style.fontFamily = 'sans-serif';
        btn.style.fontSize = '24px';
        btn.innerText = 'Base';
        btn.onclick = () => openPicker('main', 'set');
        container.appendChild(btn);
    } else {
        container.appendChild(createBox(state.main, 'main', 0, 'main-box', true));
    }
}

function renderTop() {
    const container = document.getElementById('area-top');
    container.innerHTML = '';
    if (!state.main) {
        container.innerHTML = `<button class="add-btn btn-top-full empty-slot" onclick="triggerMainHint()">+</button>`;
        return; 
    }
    if (state.top.length === 0) {
        container.innerHTML = `<button class="add-btn btn-top-full empty-slot" onclick="openPicker('top', 'push')">+</button>`;
    } else if (state.top.length === 1) {
        let bl = document.createElement('button'); bl.className = 'add-btn btn-40-70'; bl.innerText = '+';
        bl.onclick = () => openPicker('top', 'pushLeft');
        let br = document.createElement('button'); br.className = 'add-btn btn-40-70'; br.innerText = '+';
        br.onclick = () => openPicker('top', 'pushRight');
        container.appendChild(bl);
        container.appendChild(createBox(state.top[0], 'top', 0, 'mod-box box-70'));
        container.appendChild(br);
    } else {
        container.appendChild(createBox(state.top[0], 'top', 0, 'mod-box box-75-70'));
        container.appendChild(createBox(state.top[1], 'top', 1, 'mod-box box-75-70'));
    }
}

function renderSide() {
    const container = document.getElementById('area-side');
    container.innerHTML = '';
    if (!state.main) {
        container.innerHTML = `<button class="add-btn btn-side-full empty-slot" onclick="triggerMainHint()">+</button>`;
        return; 
    }
    if (state.side.length === 0) {
        container.innerHTML = `<button class="add-btn btn-side-full empty-slot" onclick="openPicker('side', 'push')">+</button>`;
    } else if (state.side.length === 1) {
        let bt = document.createElement('button'); bt.className = 'add-btn btn-70-40'; bt.innerText = '+';
        bt.onclick = () => openPicker('side', 'pushTop');
        let bb = document.createElement('button'); bb.className = 'add-btn btn-70-40'; bb.innerText = '+';
        bb.onclick = () => openPicker('side', 'pushBot');
        container.appendChild(bt);
        container.appendChild(createBox(state.side[0], 'side', 0, 'mod-box box-70'));
        container.appendChild(bb);
    } else {
        container.appendChild(createBox(state.side[0], 'side', 0, 'mod-box box-70-75'));
        container.appendChild(createBox(state.side[1], 'side', 1, 'mod-box box-70-75'));
    }
}

function removeGlyph(area, index) {
    if (area === 'main') state = { main: null, top: [], side: [] };
    else state[area].splice(index, 1);
    renderUI();
}

function flipGlyph(area, index) {
    if (area === 'main') state.main.reversed = !state.main.reversed;
    else state[area][index].reversed = !state[area][index].reversed;
    renderUI();
}

function updateOutput() {
    if (!state.main) {
        document.getElementById('out-string').innerText = "...";
        document.getElementById('out-logogram').innerText = "";
        return;
    }
    let m = state.main.char;
    m = (state.top.length > 0) ? m.toUpperCase() : m.charAt(0).toUpperCase() + m.charAt(1).toLowerCase();
    if (state.main.reversed) m += "!";
    let t = "";
    if (state.top.length === 1) t += state.top[0].char + (state.top[0].reversed ? "!" : "");
    else if (state.top.length === 2) {
        t += state.top[1].char + (state.top[1].reversed ? "!" : ""); 
        t += state.top[0].char + (state.top[0].reversed ? "!" : ""); 
    }
    let s = "";
    if (state.side.length === 1) s += state.side[0].char + (state.side[0].reversed ? "!" : "");
    else if (state.side.length === 2) {
        s += state.side[0].char + (state.side[0].reversed ? "!" : ""); 
        s += state.side[1].char + (state.side[1].reversed ? "!" : ""); 
    }
    let finalStr = m + t + s;
    document.getElementById('out-string').innerText = finalStr;
    document.getElementById('out-logogram').innerText = finalStr;
}

function openPicker(area, action) {
    activeTarget = { area, action };
    document.getElementById('glyph-picker-modal').style.display = 'block';
}

function closePicker() {
    document.getElementById('glyph-picker-modal').style.display = 'none';
    activeTarget = null;
}

document.querySelectorAll('#glyphTable td').forEach(td => {
    td.addEventListener('click', () => {
        let span = td.querySelector('span.yiv-font');
        if (span && activeTarget) {
            let val = span.innerText;
            if (activeTarget.area !== 'main') val = val.toLowerCase();
            let obj = { char: val, reversed: false };
            if (activeTarget.area === 'main') state.main = obj;
            else if (activeTarget.action === 'push') state[activeTarget.area].push(obj);
            else if (activeTarget.action === 'pushLeft' || activeTarget.action === 'pushTop') state[activeTarget.area].unshift(obj);
            else if (activeTarget.action === 'pushRight' || activeTarget.action === 'pushBot') state[activeTarget.area].push(obj);
            closePicker();
            renderUI();
        }
    });
});

window.onclick = (e) => { if (e.target == document.getElementById('glyph-picker-modal')) closePicker(); }

document.getElementById('btn-clear').addEventListener('click', () => {
    state = { main: null, top: [], side: [] };
    renderUI();
});

document.getElementById('btn-copy').addEventListener('click', (e) => {
    let text = document.getElementById('out-string').innerText;
    if (text !== "...") {
        navigator.clipboard.writeText(text);
        e.target.innerText = "Copied!";
        setTimeout(() => e.target.innerText = "Copy", 2000);
    }
});
</script>
